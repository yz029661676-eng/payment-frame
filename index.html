// ... (שאר הקוד נשאר אותו דבר) async function receiveMessage(event) { const data = event.data; if (!data || !data.Name) return; // ... (טיפול ב-Height) if (data.Name === "TransactionResponse") { const res = data.Value; // --- חילוץ ה-ID בצורה חכמה --- let extractedId = "unknown"; try { // 1. נסיון ראשון: מהכתובת המלאה של הדף המארח const ref = document.referrer; if (ref) { // מחפשים מספרים בסוף הכתובת (למשל records/12345) const matches = ref.match(/\/(\d+)(\/|$|\?)/); if (matches && matches[1]) { extractedId = matches[1]; } } } catch (e) { console.log("Could not parse ID from referrer"); } // ---------------------------- if (res.Status === "Error") { msg.textContent = res.Message || "שגיאה"; saveBtn.disabled = false; return; } const token = res.Token || res.CardToken || res.CreditToken || ""; try { const response = await fetch(MAKE_WEBHOOK, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ token: token, origami_id: extractedId, // ה-ID שחילצנו מהכתובת של אוריגמי response: res, debug_referrer: document.referrer // ליתר ביטחון, כדי שתראי במייק מאיפה זה בא }) }); if (!response.ok) throw new Error("Webhook failed"); msg.textContent = "נשמר בהצלחה"; } catch (err) { msg.textContent = "הטוקן נוצר, אך השליחה נכשלה"; } saveBtn.disabled = false; } }
